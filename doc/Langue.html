    # File Parametre/Langue.rb, line 299
def enregistreTmp( unChemin )
    File.open( unChemin , "wb" ){ |f| f.write( Marshal.dump(@dictionnaire) ) }
    @dictionnaire = Marshal.load( File.binread( unChemin ) )
end
